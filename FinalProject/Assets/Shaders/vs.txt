#version 330 core
              
layout (location = 0) in vec3 position;
layout (location = 1) in vec3 color;

uniform mat4 u_projection;

out vec3 out_position;
out vec3 Color;

void main()
{
    gl_Position = u_projection * vec4(position, 1.0);
    out_position = gl_Position.xyz;
    Color = color;
}
